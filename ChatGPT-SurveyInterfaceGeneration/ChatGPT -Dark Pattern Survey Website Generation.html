    JavaScript for Countdown Timer (Simplified Example)

    document.querySelectorAll('.countdown').forEach(function(timer) {
    // Set the countdown time here
    let countdownTime = 15 * 60 * 1000; // 15 minutes in milliseconds
    let endTime = new Date().getTime() + countdownTime;

    let interval = setInterval(function() {
        let now = new Date().getTime();
        let distance = endTime - now;

        let minutes = Math.floor((distance % (1000 * 60 * 60)) / (1000 * 60));
        let seconds = Math.floor((distance % (1000 * 60)) / 1000);

        timer.innerHTML = minutes + ":" + (seconds < 10 ? "0" : "") + seconds;

        if (distance < 0) {
            clearInterval(interval);
            timer.innerHTML = "EXPIRED";
        }
    }, 1000);
});
